\phi_{\textsf{license}}(x)

\lhd'(x,y) 
  &:= \lhd(x,y) \\


\alpha'(x,y) 
  &:= \bigl(\alpha(x,y)\land \neg \textsf{deassoc}(x,y)\bigr) \lor \textsf{reassoc}(x,y) \\

  &:= 
    \begin{cases}
	 \alpha(x,y)\land \textsf{delete\_V}(y)

	 \alpha(x,y)\land \big(\textsf{delete\_V}(y) \lor \big(\textsf{survive\_V}(y) \land \exists z \ \textsf{higher}(z,x))\big)

    \end{cases}
\\

  &:=
  \begin{cases}
    \bot
    & \text{; for Engenni} \\

    \textsf{survive\_V}(y)\land
    \exists z\,(\alpha(x,z)\land \textsf{delete\_V}(z))
    & \text{; for Estako} \\

    \textsf{survive\_V}(y)\land
\exists z \ \textsf{higher}(x,z)    & \text{; for Ogoja Yala}
  \end{cases}
\end{aligned}
